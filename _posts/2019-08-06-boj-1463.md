chk[n]=1을 했어야하는데 chk[n]==1해서 시간초과 났던 문제 금방 고치긴했지만..

### [boj 1463](https://www.acmicpc.net/problem/1463)

```c
#include<stdio.h>
#include<algorithm>
#define MAX 1000000

using namespace std;
int chk[MAX],dt[MAX];
int dp(int n) {
	if (n == 1) return 0;
	if (chk[n] == 1) return dt[n];
	chk[n] = 1;
	if(n>1)return dt[n]=min(min(n % 3 == 0 ? dp(n / 3) : MAX, n % 2 == 0 ? dp(n / 2) : MAX), dp(n - 1))+1;
}
int main() {
	int n;
	scanf("%d", &n);
	printf("%d\n", dp(n));
	return 0;
}
```
